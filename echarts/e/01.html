# 5 分钟上手 ECharts

# 获取 ECharts

你可以通过以下几种方式获取 Apache EChartsTM。

  • 从 Apache ECharts 官网下载界面 (opens new window) 获取官方源码包后构建。
  • 在 ECharts 的 GitHub (opens new window) 获取。
  • 通过 npm 获取 echarts，npm install echarts --save，详见“[在 webpack 中使用 echarts](https://echarts.apache.org/zh/tutorial.html#在 webpack 中使用 ECharts)”
  • 通过 jsDelivr (opens new window) 等 CDN 引入

<script>
import echarts from "echarts";
import 'echarts/map/js/china.js'



export default {
  name: "text",

  mounted() {
    this.chinaConfigure(); /*在mounted的声明周期进行echarts的挂载*/
  },

  methods:{
    getRowClass({ row, column, rowIndex, columnIndex }) {
      return "background:#3f5c6d2c; color:#ccc;";
    },
    chinaConfigure() {
      let myChart=echarts.init(document.getElementById('mmap')); //这里是为了获得容器所在位置
      window.onresize = myChart.resize;
      myChart.setOption({ // 进行相关配置
        //backgroundColor: "#02AFDB",

        title: {
          text: '地图',
          subtext: 'China',
          left: 'left',
          top:'0',
          textStyle: {
            color: '#ccc'
          }
        },

        tooltip: {}, // 鼠标移到图里面的浮动提示框
        dataRange: {
          show: true,
          min: 0,
          max: 500,
          text: ['High', 'Low'],
          realtime: true,
          calculable: true,
          color: ['orangered', 'yellow', 'purple']
        },

        geo: { // 这个是重点配置区
          map: 'china', // 表示中国地图
          //是否开启鼠标缩放和平移漫游
          roam: true,
          //高亮状态下的多边形和标签样式
          emphasis: {
            show: true
          },
          label: {
            normal: {
              show: true, // 是否显示对应地名
              textStyle: {
                color: '#fff'
              }
            }
          },

            normal: {
              shadowBlur: 50,

//数据转换，转换后格式：[{fromName:'cityName', toName:'cityName', coords:[[lng, lat], [lng, lat]]}, {...}]
const convertFlyData = function(data) {
  let res = [];
  for(let i=0;i<data.length;i++) {
    let dataItem = data[i];
    let toCoord = geoCoordMap[dataItem[0].name];
    let fromCoord = geoCoordMap[dataItem[1].name];
    if(fromCoord && toCoord) {
      res.push({
        fromName: dataItem[1].name,
        toName: dataItem[0].name,
        coords: [fromCoord, toCoord]
      });
    }
  }
  return res;
};
